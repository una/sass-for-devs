Variables

.scss

							$font--sans: 'Open Sans', sans-serif;
$pink: #c69;

body {
  font-family: $font--sans;
}

h1 {
  color: $pink;
}

.css output

						body {
  font-family: "Open Sans", sans-serif;
}

h1 {
  color: #c69;
}
